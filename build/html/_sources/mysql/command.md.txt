# コマンド一覧

```bash
# バックアップ
mysqldump --single-transaction -u root -p sample > sample.dmp

# リストア
mysql -u root -ppassword -D sample < sample.dmp

# ログイン DB指定なし
mysql -h localhost -u root -ppassw0rd

# ログイン DB指定
mysql -h localhost -u root -ppassw0rd sample

# データベース一覧
show databases;

# テーブル一覧
show tables;

```
